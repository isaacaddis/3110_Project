Module Main

module Main: sig .. end

The main entry point for the game interface.

val quit : unit -> 'a

quit () prints a message and then exits the game.

val format_pts : int -> string

format_pts p formats the points from a players hand.

val print_st : bool -> State.t -> unit

print_st b st prints the state, and changes depending on if st is the players turn or the dealers turn with bool b.

val game_msg : State.t -> unit -> string

game_msg s prints the game message for state s and returns the user input afterwards.

val game_double : State.t -> Controller.win_condition

game_double s plays out game with state s after doubling down.

val game_loop : int -> bool -> State.t -> Controller.win_condition

game_loop b p s loops the game with state s and boolean for player turn p, with player bet b.

val read_bet : string -> int

read_bet inp runs the game state if inp is a proper integer bet, and asks for reinput if the inp is not appropriate.

val print_result : float -> unit

print_result p prints the result depending on the dollars that you gained or lost.

val check_bankrupt : int -> unit

check_bankrupt m quites the game if m is equal to 0.

val play_game : State.t -> unit -> unit

play_game b s runs the Blackjack game with state s.

val main : unit -> unit

main () runs the game, and gives the player instructions.
